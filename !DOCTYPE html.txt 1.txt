    </div>

    <script>
        // Global variables
        let recognition = null;
        let mediaRecorder = null;
        let audioChunks = [];
        let isListening = false;
        let isRecordingForWhisper = false;
        let speechSynth = window.speechSynthesis;
        let currentPage = 'home';
        let recognitionAttempts = 0;
        let lastRecognitionConfidence = 1;
        let contacts = [];
        let filteredContacts = [];
        let selectedContact = null;
        
        // Settings
        let settings = {
            pauseTimeout: 2,
            speechSpeed: 1.0,
            speakVoiceCommands: true,
            autoEnhanced: true,
            apiKey: ''
        };

        // Sample contacts data
        const sampleContacts = [
            { name: "John Smith", phone: "(555) 123-4567" },
            { name: "Sarah Johnson", phone: "(555) 234-5678" },
            { name: "Mike Wilson", phone: "(555) 345-6789" },
            { name: "Emily Davis", phone: "(555) 456-7890" },
            { name: "David Brown", phone: "(555) 567-8901" },
            { name: "Lisa Anderson", phone: "(555) 678-9012" },
            { name: "Chris Taylor", phone: "(555) 789-0123" },
            { name: "Jessica Martin", phone: "(555) 890-1234" },
            { name: "Robert Garcia", phone: "(555) 901-2345" },
            { name: "Amanda Rodriguez", phone: "(555) 012-3456" }
        ];

        // Touch handling for swipe detection
        let touchStartY = 0;
        let touchEndY = 0;

        // Initialize the app
        function initApp() {
            // Load settings
            loadSettings();
            
            // Check for speech recognition support
            if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
                alert('Speech recognition not supported in this browser. Please try Chrome, Edge, or Safari.');
                return;
            }

            // Initialize speech recognition
            setupSpeechRecognition();
            
            // Initialize contacts
            initializeContacts();
            
            // Setup swipe detection
            setupSwipeDetection();
            
            // Check if we should show enhanced section
            updateEnhancedSectionVisibility();

            console.log('Leslie Speech Assistant initialized');
        }

        function setupSpeechRecognition() {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            recognition = new SpeechRecognition();
            
            recognition.continuous = true;
            recognition.interimResults = true;
            recognition.lang = 'en-US';
            
            recognition.onstart = function() {
                isListening = true;
                updateListenButton();
                updateStatus('🎤 Listening... Speak now!');
                recognitionAttempts++;
            };
            
            recognition.onend = function() {
                isListening = false;
                updateListenButton();
                updateStatus('Ready to listen - Click "START TALKING" below');
                
                // Show enhanced suggestion if recognition struggled
                if (settings.autoEnhanced && recognitionAttempts > 2 && lastRecognitionConfidence < 0.8) {
                    showEnhancedSuggestion();
                }
            };
            
            recognition.onerror = function(event) {
                console.error('Speech recognition error:', event.error);
                isListening = false;
                updateListenButton();
                updateStatus('Error: ' + event.error + '. Try Enhanced Recognition for better accuracy.');
                
                // Show enhanced section on error
                if (settings.autoEnhanced) {
                    showEnhancedSuggestion();
                }
            };
            
            recognition.onresult = function(event) {
                let finalTranscript = '';
                let interimTranscript = '';
                let confidence = 0;
                
                for (let i = event.resultIndex; i < event.results.length; i++) {
                    const transcript = event.results[i][0].transcript;
                    confidence = event.results[i][0].confidence || 1;
                    
                    if (event.results[i].isFinal) {
                        finalTranscript += transcript;
                        lastRecognitionConfidence = confidence;
                    } else {
                        interimTranscript += transcript;
                    }
                }
                
                const recognizedTextEl = document.getElementById('recognizedText');
                if (finalTranscript) {
                    recognizedTextEl.textContent += finalTranscript + ' ';
                    processVoiceCommand(finalTranscript);
                    
                    // Auto-scroll to bottom
                    const speechContent = recognizedTextEl.parentElement;
                    speechContent.scrollTop = speechContent.scrollHeight;
                    
                    // Show enhanced suggestion for low confidence
                    if (settings.autoEnhanced && confidence < 0.7) {
                        showEnhancedSuggestion();
                    }
                }
                
                // Show interim results
                if (interimTranscript) {
                    updateStatus('🔄 Recognizing: ' + interimTranscript);
                } else if (finalTranscript) {
                    updateStatus('✅ Recognized: ' + finalTranscript);
                }
            };
        }

        // Initialize contacts
        function initializeContacts() {
            contacts = [...sampleContacts];
            filteredContacts = [...contacts];
            displayContacts();
        }

        function displayContacts() {
            const contactsList = document.getElementById('contactsList');
            contactsList.innerHTML = '';
            
            if (filteredContacts.length === 0) {
                contactsList.innerHTML = '<div style="padding: 16px; text-align: center; color: #666;">No contacts found</div>';
                return;
            }
            
            filteredContacts.forEach((contact, index) => {
                const contactEl = document.createElement('div');
                contactEl.className = 'contact-item';
                contactEl.innerHTML = `<strong>${contact.name}</strong><br><span style="color: #666; font-size: 14px;">${contact.phone}</span>`;
                contactEl.onclick = () => selectContact(contact);
                contactsList.appendChild(contactEl);
            });
        }

        function filterContacts(searchTerm) {
            if (!searchTerm.trim()) {
                filteredContacts = [...contacts];
            } else {
                filteredContacts = contacts.filter(contact => 
                    contact.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
                    contact.phone.includes(searchTerm)
                );
            }
            displayContacts();
        }

        function selectContact(contact) {
            selectedContact = contact;
            
            // Update UI
            document.getElementById('selectedContactLabel').style.display = 'block';
            document.getElementById('phoneNumberInput').style.display = 'block';
            document.getElementById('phoneNumberInput').value = `${contact.name} - ${contact.phone}`;
            document.getElementById('makeCallButton').style.display = 'block';
            
            // Highlight selected contact
            const contactItems = document.querySelectorAll('.contact-item');
            contactItems.forEach(item => item.classList.remove('selected'));
            event.target.classList.add('selected');
        }

        function makeCall() {
            if (selectedContact) {
                // On mobile, this will open the phone app
                window.location.href = `tel:${selectedContact.phone}`;
            }
        }

        // Swipe detection
        function setupSwipeDetection() {
            const container = document.querySelector('.container');
            
            container.addEventListener('touchstart', function(e) {
                touchStartY = e.changedTouches[0].screenY;
            });

            container.addEventListener('touchend', function(e) {
                touchEndY = e.changedTouches[0].screenY;
                handleSwipe();
            });
        }

        function handleSwipe() {
            const swipeDistance = touchStartY - touchEndY;
            const minSwipeDistance = 100;
            
            // Check if swipe started from bottom 100px of screen
            const isFromBottom = touchStartY > window.innerHeight - 100;
            
            if (isFromBottom && swipeDistance > minSwipeDistance) {
                // Swipe up from bottom - show weather
                showWeather();
            }
        }

        function showWeather() {
            const weatherDisplay = document.getElementById('weatherDisplay');
            const weatherContent = document.getElementById('weatherContent');
            
            // Simple weather simulation
            const currentTime = new Date();
            const temperature = Math.floor(Math.random() * 30) + 60; // 60-90°F
            const conditions = ['Sunny', 'Partly Cloudy', 'Cloudy', 'Light Rain', 'Clear'][Math.floor(Math.random() * 5)];
            
            weatherContent.innerHTML = `
                <strong>Current Weather</strong><br>
                Temperature: ${temperature}°F<br>
                Conditions: ${conditions}<br>
                Time: ${currentTime.toLocaleTimeString()}<br><br>
                <em>This is simulated weather data for demo purposes</em>
            `;
            
            weatherDisplay.classList.remove('hidden');
        }

        function hideWeather() {
            document.getElementById('weatherDisplay').classList.add('hidden');
        }

        // Enhanced Recognition Functions
        async function startEnhancedRecognition() {
            if (!settings.apiKey) {
                alert('Please enter your OpenAI API key in Settings to use Enhanced Recognition.');
                showPage('settings');
                return;
            }

            if (isRecordingForWhisper) {
                stopEnhancedRecognition();
                return;
            }

            try {
                const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                setupMediaRecorder(stream);
                
                mediaRecorder.start();
                isRecordingForWhisper = true;
                updateEnhancedButton();
                updateEnhancedStatus('🔴 Recording... Speak clearly for up to 30 seconds');
                document.getElementById('recordingIndicator').classList.add('show');
                
                // Auto-stop after 30 seconds
                setTimeout(() => {
                    if (isRecordingForWhisper) {
                        stopEnhancedRecognition();
                    }
                }, 30000);
                
            } catch (error) {
                console.error('Error accessing microphone:', error);
                updateEnhancedStatus('Error: Could not access microphone');
            }
        }

        function setupMediaRecorder(stream) {
            audioChunks = [];
            mediaRecorder = new MediaRecorder(stream);
            
            mediaRecorder.ondataavailable = function(event) {
                audioChunks.push(event.data);
            };
            
            mediaRecorder.onstop = function() {
                const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });
                sendToWhisper(audioBlob);
                
                // Stop all tracks to release microphone
                stream.getTracks().forEach(track => track.stop());
            };
        }

        function stopEnhancedRecognition() {
            if (mediaRecorder && isRecordingForWhisper) {
                mediaRecorder.stop();
                isRecordingForWhisper = false;
                updateEnhancedButton();
                document.getElementById('recordingIndicator').classList.remove('show');
                updateEnhancedStatus('⏳ Processing audio with OpenAI Whisper...');
            }
        }

        async function sendToWhisper(audioBlob) {
            const formData = new FormData();
            formData.append('file', audioBlob, 'audio.wav');
            formData.append('model', 'whisper-1');
            formData.append('language', 'en');

            try {
                const response = await fetch('https://api.openai.com/v1/audio/transcriptions', {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${settings.apiKey}`
                    },
                    body: formData
                });

                if (!response.ok) {
                    throw new Error(`API Error: ${response.status} - ${response.statusText}`);
                }

                const result = await response.json();
                handleWhisperResult(result.text);
                
            } catch (error) {
                console.error('Whisper API error:', error);
                updateEnhancedStatus('❌ Error: ' + error.message);
                
                if (error.message.includes('401')) {
                    updateEnhancedStatus('❌ Error: Invalid API key. Please check your settings.');
                } else if (error.message.includes('429')) {
                    updateEnhancedStatus('❌ Error: Rate limit exceeded. Please try again later.');
                }
            }
        }

        function handleWhisperResult(text) {
            if (text && text.trim()) {
                const recognizedTextEl = document.getElementById('recognizedText');
                recognizedTextEl.textContent += '[Enhanced] ' + text + ' ';
                
                updateEnhancedStatus('✅ Enhanced recognition complete! Result added to text above.');
                processVoiceCommand(text);
                
                // Scroll to show new text
                const speechContent = recognizedTextEl.parentElement;
                speechContent.scrollTop = speechContent.scrollHeight;
            } else {
                updateEnhancedStatus('No speech detected. Please try speaking closer to the microphone.');
            }
        }

        function showEnhancedSuggestion() {
            const enhancedSection = document.getElementById('enhancedSection');
            enhancedSection.classList.add('show');
            updateEnhancedStatus('💡 Try Enhanced Recognition for better accuracy with unclear speech');
        }

        function updateEnhancedButton() {
            const button = document.getElementById('enhancedButton');
            if (isRecordingForWhisper) {
                button.textContent = '⏹️ STOP ENHANCED RECORDING';
                button.classList.add('recording');
            } else {
                button.textContent = '🔬 START ENHANCED RECOGNITION';
                button.classList.remove('recording');
            }
        }

        function updateEnhancedStatus(message) {
            const statusEl = document.getElementById('enhancedStatus');
            statusEl.textContent = message;
        }

        function updateEnhancedSectionVisibility() {
            const hasApiKey = settings.apiKey && settings.apiKey.length > 0;
            const enhancedSection = document.getElementById('enhancedSection');
            
            if (hasApiKey) {
                enhancedSection.style.display = 'block';
            } else {
                enhancedSection.style.display = 'none';
            }
        }

        // Page navigation
        function showPage(pageId) {
            // Stop any active recording when changing pages
            if (isListening) {
                stopListening();
            }
            if (isRecordingForWhisper) {
                stopEnhancedRecognition();
            }
            
            // Hide all pages
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => page.classList.remove('active'));
            
            // Show selected page
            const targetPage = document.getElementById(pageId + 'Page');
            if (targetPage) {
                targetPage.classList.add('active');
                currentPage = pageId;
            }
        }

        // Special function for starting talking from home
        function startTalkingFromHome() {
            showPage('speech');
            // Small delay to ensure page is shown, then start listening
            setTimeout(() => {
                startListening();
            }, 100);
        }

        // Speech recognition functions
        function toggleListening() {
            if (isListening) {
                stopListening();
            } else {
                startListening();
            }
        }

        function startListening() {
            if (recognition && !isListening) {
                try {
                    recognition.start();
                } catch (error) {
                    console.error('Error starting recognition:', error);
                    updateStatus('Error starting recognition. Please try again.');
                }
            }
        }

        function stopListening() {
            if (recognition && isListening) {
                recognition.stop();
            }
        }

        function updateListenButton() {
            const button = document.getElementById('listenButton');
            if (isListening) {
                button.textContent = 'STOP TALKING';
                button.className = 'control-btn btn-stop recording';
            } else {
                button.textContent = 'START TALKING';
                button.className = 'control-btn btn-listen';
            }
        }

        function updateStatus(message) {
            const statusEl = document.getElementById('statusText');
            if (statusEl) {
                statusEl.textContent = message;
            }
        }

        // Text-to-speech
        function speakText(text) {
            if (!text) {
                text = document.getElementById('largeTextInputField').value.trim();
            }
            
            if (!text) {
                alert('Please enter some text to speak');
                return;
            }
            
            // Cancel any ongoing speech
            speechSynth.cancel();
            
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.rate = settings.speechSpeed;
            utterance.pitch = 1;
            utterance.volume = 1;
            
            utterance.onstart = function() {
                updateTextInputStatus('🗣️ Speaking text...');
            };
            
            utterance.onend = function() {
                updateTextInputStatus('');
            };
            
            utterance.onerror = function(event) {
                console.error('Speech synthesis error:', event);
                updateTextInputStatus('❌ Error speaking text');
            };
            
            speechSynth.speak(utterance);
        }

        function updateTextInputStatus(message) {
            const statusEl = document.getElementById('textInputStatusText');
            if (statusEl) {
                statusEl.textContent = message;
                statusEl.style.display = message ? 'block' : 'none';
            }
        }

        function clearText() {
            document.getElementById('largeTextInputField').value = '';
        }

        // Voice commands processing
        function processVoiceCommand(text) {
            const lowerText = text.toLowerCase();
            
            if (lowerText.includes('hey leslie')) {
                if (lowerText.includes('what time is it')) {
                    const time = new Date().toLocaleTimeString();
                    respondToCommand('The current time is ' + time);
                } else if (lowerText.includes('what\'s the date') || lowerText.includes('what is the date')) {
                    const date = new Date().toLocaleDateString('en-US', { 
                        weekday: 'long', 
                        year: 'numeric', 
                        month: 'long', 
                        day: 'numeric' 
                    });
                    respondToCommand('Today is ' + date);
                } else if (lowerText.includes('start talking')) {
                    if (currentPage !== 'speech') {
                        startTalkingFromHome();
                    }
                    respondToCommand('Starting speech recognition');
                } else if (lowerText.includes('speak this')) {
                    respondToCommand('You said: ' + text);
                } else if (lowerText.includes('clear text')) {
                    document.getElementById('recognizedText').textContent = 'Your recognized speech will appear here...';
                    respondToCommand('Text cleared');
                } else if (lowerText.includes('enhanced mode')) {
                    showEnhancedSuggestion();
                    respondToCommand('Enhanced recognition mode is now available. Click the Enhanced Recognition button.');
                } else if (lowerText.includes('weather')) {
                    showWeather();
                    respondToCommand('Showing weather information');
                } else if (lowerText.includes('go home')) {
                    showPage('home');
                    respondToCommand('Going to home screen');
                } else if (lowerText.includes('help')) {
                    showPage('commands');
                    respondToCommand('Showing commands list');
                } else {
                    respondToCommand('I heard you say Hey Leslie, but I didn\'t understand the command. Say Hey Leslie help for available commands.');
                }
            }
        }

        function respondToCommand(message) {
            if (settings.speakVoiceCommands) {
                const utterance = new SpeechSynthesisUtterance(message);
                utterance.rate = settings.speechSpeed;
                speechSynth.speak(utterance);
            }
            updateStatus('Leslie: ' + message);
        }

        // Settings functions
        function loadSettings() {
            const saved = localStorage.getItem('leslieSettings');
            if (saved) {
                settings = { ...settings, ...JSON.parse(saved) };
            }
            
            // Update UI with loaded settings
            document.getElementById('pauseTimeoutSlider').value = settings.pauseTimeout;
            document.getElementById('speechSpeedSlider').value = settings.speechSpeed;
            document.getElementById('speakVoiceCommandsSwitch').checked = settings.speakVoiceCommands;
            document.getElementById('autoEnhancedSwitch').checked = settings.autoEnhanced;
            document.getElementById('apiKeyInput').value = settings.apiKey;
            
            updatePauseTimeout(settings.pauseTimeout);
            updateSpeechSpeed(settings.speechSpeed);
            updateVoiceCommandSetting(settings.speakVoiceCommands);
            updateAutoEnhanced(settings.autoEnhanced);
        }

        function saveSettings() {
            localStorage.setItem('leslieSettings', JSON.stringify(settings));
        }

        function saveApiKey(key) {
            settings.apiKey = key;
            saveSettings();
            updateEnhancedSectionVisibility();
        }

        function updatePauseTimeout(value) {
            settings.pauseTimeout = parseInt(value);
            document.getElementById('pauseTimeoutValue').textContent = value + ' seconds';
            saveSettings();
        }

        function updateSpeechSpeed(value) {
            settings.speechSpeed = parseFloat(value);
            const speedText = value == 1.0 ? 'Normal (1.0x)' : 'Speed (' + value + 'x)';
            document.getElementById('speechSpeedValue').textContent = speedText;
            saveSettings();
        }

        function updateVoiceCommandSetting(checked) {
            settings.speakVoiceCommands = checked;
            document.getElementById('speakVoiceCommandsLabel').textContent = 
                'Voice Command Speech: ' + (checked ? 'ON' : 'OFF');
            saveSettings();
        }

        function updateAutoEnhanced(checked) {
            settings.autoEnhanced = checked;
            saveSettings();
        }

        // Initialize app when page loads
        window.addEventListener('load', initApp);

        // Handle page visibility changes
        document.addEventListener('visibilitychange', function() {
            if (document.hidden) {
                if (isListening) {
                    stopListening();
                }
                if (isRecordingForWhisper) {
                    stopEnhancedRecognition();
                }
            }
        });
    </script>
